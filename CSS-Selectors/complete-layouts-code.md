
# 📐 بانک کدهای جامع سیستم‌های طرح‌بندی (Jonas Layout Master Reference)

این فایل حاوی تمام کدهای فنی، ویژگی‌ها، مقادیر ممکن و ترفندهای اجرایی سه نسل چیدمان در وب است.

---

## 🪵 ۱. آلبوم کدهای سیستم قدیمی فلوت (Float Layout)

### 💻 کدهای بدنه و هک کلیرفیکس
```css
/* ۱. هک کلیرفیکس (Clearfix Hack) - واجب‌ترین کد برای باکس مادر فلوت‌ها */
.clearfix::after {
  content: "";
  display: table;
  clear: both; /* تمام اثرات فلوت‌های قبلی را پاک می‌کند تا ارتفاع باکس مادر صفر نشود */
}

/* ۲. باکس مادر المان‌های فلوت شده */
.container-float {
  width: 1140px;
  margin: 0 auto; /* وسط‌چین کردن کل باکس در صفحه */
}

/* ۳. ستون فلوت شده به چپ (فرزند) */
.col-left {
  float: left;     /* هل دادن به چپ */
  width: 33.33%;   /* ساخت ساختار ۳ ستونه */
  padding: 20px;
  box-sizing: border-box;
}

/* ۴. ستون فلوت شده به راست (فرزند) */
.col-right {
  float: right;    /* هل دادن به راست */
  width: 50%;      /* ساخت ساختار ۲ ستونه */
  padding: 20px;
}

/* ۵. خنثی‌سازی دستی برای المان‌های بعدی */
.next-section {
  clear: both;     /* اجازه نمی‌دهد این بخش زیر فلوت‌های بالایی جابه‌جا شود */
}






⛓️ ۲. آلبوم کدهای سیستم فلکس‌باکس (Flexbox Reference)
💻 کدهای مخصوص باکس مادر (Flex Container)
CSS
.flex-master-box {
  display: flex; /* روشن کردن موتور فلکس‌باکس */

  /* جهت محور اصلی (Main Axis) */
  flex-direction: row;            /* ردیفی از چپ به راست (پیش‌فرض) */
  /* flex-direction: column; */     /* ستونی از بالا به پایین */
  /* flex-direction: row-reverse; */ /* ردیفی از راست به چپ */

  /* تراز کردن المان‌ها روی محور اصلی */
  justify-content: flex-start;    /* چسبیدن به اول خط (پیش‌فرض) */
  /* justify-content: flex-end; */      /* چسبیدن به آخر خط */
  /* justify-content: center; */        /* وسط‌چین کردن المان‌ها */
  /* justify-content: space-between; */ /* چسبیدن به لبه‌ها و تقسیم مساوی فضای خالی */
  /* justify-content: space-around; */  /* فاصله مساوی دور تا دور المان‌ها */
  /* justify-content: space-evenly; */  /* فاصله کاملاً برابر از لبه‌ها و از هم */

  /* تراز کردن المان‌ها روی محور فرعی (عمودی) */
  align-items: stretch;           /* کشیده شدن هم‌قد با بلندترین المان (پیش‌فرض) */
  /* align-items: center; */            /* تراز دقیق در مرکز عمودی */
  /* align-items: flex-start; */        /* چسبیدن به سقف باکس */
  /* align-items: flex-end; */          /* چسبیدن به کف باکس */

  /* شکستن خطوط در صورت کمبود جا */
  flex-wrap: nowrap;              /* فشرده شدن همه در یک خط (پیش‌فرض) */
  /* flex-wrap: wrap; */                /* سقوط به خط بعدی در صورت نبود جا */

  /* فاصله‌دهی هوشمند بین فرزندان */
  gap: 3.2rem;                    /* ۳۲ پیکسل فاصله بین تک‌تک المان‌ها */
  /* row-gap: 2rem; */            /* فقط فاصله بین ردیف‌ها */
  /* column-gap: 4rem; */         /* فقط فاصله بین ستون‌ها */
}
💻 کدهای مخصوص المان‌های فرزند (Flex Items)
CSS
.flex-child-box {
  /* میزان رشد المان برای پر کردن فضای خالی (پیش‌فرض: 0) */
  flex-grow: 1; 

  /* میزان مچاله شدن المان در صورت کمبود جا (پیش‌فرض: 1) */
  flex-shrink: 0; /* به هیچ وجه مچاله یا کوچک‌تر از حدش نشود */

  /* اندازه پایه اولیه المان قبل از رشد یا مچاله شدن */
  flex-basis: 25rem; 

  /* شورت‌کد استاندارد جوناس (ترکیب هر سه مورد بالا) */
  /* فرمول: flex: [grow] [shrink] [basis]; */
  flex: 1 0 25rem;

  /* تراز عمودی اختصاصی فقط و فقط برای همین یک المان */
  align-self: flex-end; /* این المان خلاف بقیه به کف باکس می‌چسبد */

  /* جابه‌جایی ترتیبی المان بدون دست زدن به کدهای HTML */
  order: -1; /* این آیتم را می‌آورد اول از همه نشان می‌دهد */
}





💻 کدهای مخصوص المان‌های فرزند (Flex Items)
CSS

.grid-master-box {
  display: grid; /* روشن کردن موتور گرید */

  /* ساخت ستون‌ها با واحد fr و تابع repeat */
  grid-template-columns: repeat(3, 1fr); /* ساخت ۳ ستون کاملاً مساوی و داینامیک */
  /* grid-template-columns: 200px 1fr 2fr; ستون اول ثابت، دومی ۱ سهم، سومی ۲ سهم */
  /* grid-template-columns: repeat(auto-fit, minmax(25rem, 1fr)); جادوی ریسپانسیو اتوماتیک بدون مدیاکوئری */

  /* ساخت ردیف‌ها */
  grid-template-rows: auto; /* ردیف‌ها به صورت خودکار اندازه محتوا را می‌گیرند */
  /* grid-auto-rows: 25rem; ردیف‌های جدیدی که خودکار اضافه می‌شوند همگی ۲۵ رم قد داشته باشند */

  /* ایجاد فاصله بین سلول‌های جدول (هم افقی هم عمودی) */
  gap: 4.8rem; 
  /* row-gap: 3rem; */
  /* column-gap: 2rem; */

  /* تراز کردن محتوای تمام سلول‌ها به صورت یکجا */
  justify-items: center; /* تراز افقی داخل سلول‌ها */
  align-items: center;   /* تراز عمودی داخل سلول‌ها */
}
💻 کدهای مخصوص المان‌های فرزند (Grid Items)
.grid-child-box {
  /* تعیین پهنای المان روی ستون‌ها (شروع / پایان) */
  grid-column-start: 1;
  grid-column-end: 3; /* از خط ۱ شروع و تا خط ۳ کشیده می‌شود (پوشش ۲ ستون) */

  /* شورت‌کد استاندارد برای پهنای ستون */
  grid-column: 1 / 3; 
  /* grid-column: span 2; ترفند ساده: به اندازه ۲ ستون پهن شو */

  /* تعیین ارتفاع المان روی ردیف‌ها */
  grid-row: 1 / 4; /* از ردیف ۱ تا ۴ کشیده می‌شود (پوشش ۳ ردیف عمودی) */
  /* grid-row: span 2; به اندازه ۲ ردیف عمودی جا بگیر */

  /* تراز اختصاصی محتوای همین یک سلول */
  justify-self: start; /* تراز افقی اختصاصی */
  align-self: center;  /* تراز عمودی اختصاصی */
}
